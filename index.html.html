// Función para limpiar datos específicos function limpiarDatosEspecificos(tipo) { const mensajes = { 'usuarios': 'eliminar TODOS los usuarios?', 'cursos': 'eliminar TODOS los cursos?', 'solicitudes': 'eliminar TODAS las solicitudes?', 'notificaciones': 'eliminar TODAS las notificaciones?' }; if (!confirmarAccion(mensajes[tipo])) { return; } try { switch(tipo) { case 'usuarios': users = []; localStorage.removeItem('users'); break; case 'cursos': cursos = []; localStorage.removeItem('cursos'); break; case 'solicitudes': solicitudes = []; localStorage.removeItem('solicitudes'); break; case 'notificaciones': notis = []; localStorage.removeItem('notis'); break; } mostrarMensaje(`✅ ${tipo.charAt(0).toUpperCase() + tipo.slice(1)} eliminados exitosamente`, 'exito', '🗑️'); cargarEstadisticas(); } catch (error) { console.error(`Error al limpiar ${tipo}:`, error); mostrarMensaje(`❌ Error al eliminar ${tipo}`, 'error', '❌'); } }
